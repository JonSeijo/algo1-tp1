
\subsection{Campo}

\aux{posNoNegativas}{pos: (\ent, \ent)}{\bool}{(prm(pos) \geq 0 \land sgd(pos) \geq 0) }

\aux{dameAncho}{c: Campo}{\ent}{prm(dimensiones(c))}
\aux{dameLargo}{c: Campo}{\ent}{sgd(dimensiones(c))}

\subsection{Drone}
% los aux del tipo drone

\aux{listaSinRepeticiones}{xs: [T]}{[T]}{[\hspace{2mm} xs[i]\hspace{3mm} |\hspace{3mm} \selector{i}{[0.. |xs|)}, xs[i] \notin xs[0..i-1] ]}

%problema (12)

\aux{posicionesPorMomento}{ds : [Drone]}{[[(\ent, \ent)]]}{[\hspace{2mm}listaVertical(d)[i]\hspace{3mm}|\hspace{3mm} \selector{i}{\rangoca{0}{|vueloRealizado(ds[0])|}}]}


\aux{listaVertical}{ds : [Drone], i : \ent}{[(\ent, \ent)]}{
[vueloRealizado(d)[i]\hspace{3mm}|\hspace{3mm} \selector{d}{ds}]}

\aux{listaSinRepetidos}{xs : [T]}{[T]}{[xs[i]\hspace{3mm}|\hspace{3mm}\selector{i}{\rangoca{0}{|xs|}},\hspace{3mm} xs[i] \notin xs\rangoca{0}{i}]}

\aux{crucesEnMomento}{xs : [(\ent, \ent)]}{[((\ent, \ent), \ent)]}{
[(x, cuenta(x, xs))\hspace{2mm}|\hspace{2mm}\selector{x}{listaSinRepetidos(xs)},cuenta(x, xs) \geq 2]}

\aux{crucesTotales}{xss : [[(\ent, \ent)]]}{[((\ent, \ent), \ent)]}{
concat(\hspace{2mm}[\hspace{1mm}crucesEnMomento(xs)\hspace{3mm}|\hspace{3mm} \selector{x}{xs}]\hspace{1mm})}

\subsection{Sistema}
% los aux del tipo sistema


\aux{noCambiaEstadoDeCultivo}{n, v : Sistema}{Bool}{
\forall \selector{i}{\rangoca{0}{dameAncho(v)}}, \forall \selector{j}{\rangoca{0}{dameLargo(v)}} \newline 
\hspace{10cm} contenido(campo(n, i, j)) == Cultivo \newline \hspace{10cm} estadoDelCultivo(v, i, j) == estadoDelCultivo(n, i, j)}


\aux{adyacentesValidas}{s : Sistema}{[(\ent, \ent)]}{[\hspace{1mm} x\hspace{1mm}|\hspace{1mm} \selector{x}{adyacentes(prm(posicionGranero(campo(s))), \newline sgd(posicionGranero(campo(s)))),\hspace{1mm} enRango(dimensiones(campo(s)), prm(x), sgd(x) ), noHayDrones(x, s)}]}
